Tip提示层 (flyer.tip) 示例 

$("#tip1").on("click", function() {
    flyer.tip("系统发生了未知错误.");
});

$("#tip2").on("click ", function() {
    flyer.tip("系统发生了未知错误.", {
        autoClose: false,
        showCancelBtn: true,
        skin: "flyer-dialog-msgError"
    });
});
$("#tip3").on("click ", function() {
    flyer.tip("<i class='fa fa-exclamation-circle flyer-icons'></i>这里的用法和其它示例是一样的，作者就不在这里再过多去展示，使用者可以自已参考前面的加上属性表里的属性拓展哦.", {
        skin: "flyer-dialog-msgError",
        autoClose: false,
        showCancelBtn: true,
        cancel: function() {
            flyer.alert("你关闭了我...");
        }
    });
});
