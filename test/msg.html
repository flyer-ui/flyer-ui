轻量消息层 (flyer.msg) 示例 

$("#msg1").on("click ", function() {
    flyer.msg("Hi,你今天笑了吗?");
});
$("#msg2").on("click ", function() {
    flyer.msg("这是一个定制高宽的消息层", {
        area: [100, 100]
    });
});
$("#msg3").on("click ", function() {
    flyer.msg("点击任意地方可以快速掉我.", {
        isModal: true,
        modalClose: true,
        area: [100, 100]
    });
});
$("#msg4").on("click ", function() {
    flyer.msg("<i class=\"fa fa-paper-plane-o flyer-icons\"></i>哼哼~~~");
});
$("#msg5").on("click ", function() {
    flyer.msg("<i class=\"fa fa-heart flyer-icons\"></i>你不点击右边的关闭按钮，我是不会走的~~~", {
        showCancelBtn: true,
        autoClose: false
    });
});
$("#msg6").on("click ", function() {
    flyer.msg("嘿嘿..", {
        showCancelBtn: true,
        autoClose: false,
        anim: "bounceInDown"
    });
});
$("#msg7").on("click ", function() {
    flyer.msg("<i class=\"fa fa-commenting-o flyer-icons\"></i>hi，大家好，这是我的新衣服，是不是觉得我今天很清纯.", {
        showCancelBtn: true,
        area: [150, 150],
        skin: "flyer-skin-default",
        cancel: function(elm) {
            flyer.msg("所以其它提示层/框有的，我也大多都有.", {
                offset: [230, 230],
                anim: "bounceInDown"
            });
        }
    });
});
$("#msg8").on("click ", function() {
    flyer.msg("<i class=\"fa fa-spinner changehovertree flyer-icons\"></i><div>正在加载中...</div>", {
        autoClose: false,
        isModal: true,
        modalClose: true,
        skin: "flyer-skin-loadding"
    });
});
