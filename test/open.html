模态框 (flyer.open) 示例 

$("#open1").on("click ", function() {
    flyer.open({
        content: "
这里的用法和其它示例是一样的，作者就不在这里再过多去展示，使用者可以自已参考前面的加上属性表里的属性拓展哦.
",
        title: "一个展现html结构的模态框",
        area: [400, 500]
    });
});
$("#open2").on("click ", function() {
    flyer.open({
        pageUrl: "tab.html",
        isModal: true,
        area: [400, 500],
        title: "显示颜色元素"
    });
});
$("#open3").on("click ", function() {
    flyer.open({
        pageUrl: "tab.html",
        isModal: true,
        area: [400, 500],
        offset: [10, 400],
        title: "显示颜色元素"
    });
});
$("#open4").on("click ", function() {
    flyer.open({
        pageUrl: "tab.html",
        skin: "flyer-skin-gray",
        isModal: true,
        area: [400, 500],
        title: "我换新衣服了"
    });
});
